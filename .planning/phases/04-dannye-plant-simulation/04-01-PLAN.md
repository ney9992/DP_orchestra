/// Конвертирует количество дней с Unix epoch (1970-01-01) в (year, month, day).
/// Используется для формирования имени архивного файла без зависимости от chrono.
fn days_to_ymd(days: u64) -> (u32, u32, u32) {
    // Алгоритм Civil Date from days: Fliegel & van Flandern / Howard Hinnant
    let z = days as i64 + 719468;
    let era = if z >= 0 { z } else { z - 146096 } / 146097;
    let doe = z - era * 146097;
    let yoe = (doe - doe/1460 + doe/36524 - doe/146096) / 365;
    let y = yoe + era * 400;
    let doy = doe - (365*yoe + yoe/4 - yoe/100);
    let mp = (5*doy + 2) / 153;
    let d = doy - (153*mp+2)/5 + 1;
    let m = if mp < 10 { mp + 3 } else { mp - 9 };
    let y = if m <= 2 { y + 1 } else { y };
    (y as u32, m as u32, d as u32)
}
